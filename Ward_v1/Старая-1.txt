  // JavaScript для управления вторым навбаром
  var prevScrollpos = window.pageYOffset;
  window.onscroll = function() {
    var currentScrollPos = window.pageYOffset;
    var navbarFirst = document.querySelector(".navbar-first");
    var navbarSecond = document.querySelector(".navbar-second");

    // Управление первым навбаром
    if (prevScrollpos > currentScrollPos) {
      navbarFirst.style.top = "0";
    } else {
      navbarFirst.style.top = "-56px";
    }

    // Второй навбар всегда фиксирован
    navbarSecond.style.top = "0";

    prevScrollpos = currentScrollPos;
